CREATE TABLE `cart` (
  `order_id` int NOT NULL AUTO_INCREMENT,
  `po_id` varchar(45) NOT NULL,
  `quantity` varchar(45) NOT NULL DEFAULT '1',
  `price` varchar(45) NOT NULL,
  `user_email` varchar(45) NOT NULL,
  `owner_email` varchar(45) NOT NULL,
  `overallprice` varchar(45) GENERATED ALWAYS AS ((`quantity` * `price`)) VIRTUAL,
  PRIMARY KEY (`order_id`)
) ENGINE=InnoDB AUTO_INCREMENT=163 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci


CREATE TABLE `favourites` (
  `fid` int NOT NULL AUTO_INCREMENT,
  `user_email` varchar(45) NOT NULL,
  `owner_email` varchar(45) NOT NULL,
  PRIMARY KEY (`fid`),
  UNIQUE KEY `owner_email_UNIQUE` (`owner_email`)
) ENGINE=InnoDB AUTO_INCREMENT=36 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci


CREATE TABLE `menu` (
  `p_id` int NOT NULL AUTO_INCREMENT,
  `p_name` varchar(45) NOT NULL,
  `p_ingredients` varchar(45) NOT NULL,
  `p_description` varchar(45) NOT NULL,
  `p_category` varchar(45) NOT NULL,
  `p_type` varchar(45) NOT NULL,
  `p_price` varchar(45) NOT NULL,
  `email` varchar(45) NOT NULL,
  `uploadPublicID` varchar(45) DEFAULT NULL,
  `uploadURL` varchar(1000) DEFAULT NULL,
  PRIMARY KEY (`p_id`)
) ENGINE=InnoDB AUTO_INCREMENT=41 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci


CREATE TABLE `order` (
  `order_id` int NOT NULL AUTO_INCREMENT,
  `po_id` varchar(45) NOT NULL,
  `user_email` varchar(45) NOT NULL,
  `quantity` varchar(45) NOT NULL,
  `sp_inst` varchar(45) DEFAULT 'N/A',
  `order_status` varchar(45) NOT NULL,
  `ordertime` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`order_id`)
) ENGINE=InnoDB AUTO_INCREMENT=122 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci


CREATE TABLE `restaurant` (
  `id` int NOT NULL AUTO_INCREMENT,
  `name` varchar(45) NOT NULL,
  `email` varchar(45) NOT NULL,
  `password` varchar(45) NOT NULL,
  `location` varchar(45) NOT NULL,
  `description` varchar(45) DEFAULT NULL,
  `contact` varchar(45) DEFAULT NULL,
  `timings` varchar(45) DEFAULT NULL,
  `delivery` varchar(45) DEFAULT NULL,
  `pickup` varchar(45) DEFAULT NULL,
  `uploadPublicID` varchar(45) DEFAULT NULL,
  `uploadURL` varchar(10000) DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `email_UNIQUE` (`email`)
) ENGINE=InnoDB AUTO_INCREMENT=19 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
